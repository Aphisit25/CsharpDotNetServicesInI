#วิธีที่ 1: ส่งค่าเป็น Command Line Argument (แนะนำ)
---
## MainProgram เรียก SignOn.exe พร้อมส่งค่า IP และ User Name ไปด้วย

Process.Start("SignOn.exe", "192.168.1.10 admin");

หรือ

ProcessStartInfo psi = new ProcessStartInfo();

psi.FileName = "SignOn.exe";

psi.Arguments = "192.168.1.10 admin";

Process.Start(psi);
